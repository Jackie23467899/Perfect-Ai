from pyspellchecker import SpellChecker def correct_spelling(query: str) -> str: spell = SpellChecker() words = query.split() corrected_words = [spell.candidates(word).pop() if spell.unknown([word]) else word for word in words] return ' '.join(corrected_words) def extract_keywords(query: str) -> str: synonyms = { "where is": ["location of", "where's"], "what is": ["define", "describe", "what's"], "who is": ["identify", "who's"], "how about": ["details on", "info about", "consider"] } query = query.lower().strip() for key, values in synonyms.items(): if any(val in query for val in values): query = query.replace(next(val for val in values if val in query), '').strip() break return query

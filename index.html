from random import choice from js import document, Element # Function to generate a nature-themed story def generate_nature_story(): characters = ['a brave knight', 'a curious cat', 'an adventurous explorer', 'a wise old wizard'] settings = ['in a magical forest', 'in a bustling city', 'on a distant planet', 'in a hidden kingdom'] conflicts = ['was searching for a lost treasure', 'discovered a mysterious portal', 'was trapped in an ancient curse', 'needed to save the world'] resolutions = ['and found what they were looking for', 'and unlocked a great secret', 'and broke the curse', 'and became a hero'] story = f"Once upon a time, {choice(characters)} {choice(settings)} {choice(conflicts)} {choice(resolutions)}." return story # Function to update the story div with a new story def update_story(): story_element = Element('output') story_element.element.innerText = generate_nature_story() # Bind the function to the button click event def handle_query(): query = document.getElementById('query').value.strip().lower() if query.startswith('tell me a story'): update_story() else: document.getElementById('output').innerText = 'Please enter a valid query.' document.getElementById('query').addEventListener('keypress', lambda e: handle_query() if e.key == 'Enter' else None)
